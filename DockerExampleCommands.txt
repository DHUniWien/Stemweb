example commands for stemweb running within a docker container

# build the docker image:
docker build -t "py2716-stemweb:dbgdocker-newcelery-redis" -f Dockerfile_newCelery .

# run the docker container:
docker run -it --name stemweb-debug-newcelery-redis -p 8000:8000 -p3000:3000 --log-driver json-file py2716-stemweb:dbgdocker-newcelery-redis 
    # -p3000:3000  ... publish EXPOSED port 3000: only used for debugging from VS code outside of the container

# open a new shell in the container:
# (useful to work within the container; the first container shell is occupied by the logging output)
docker ps         # gives us the actual container-id
docker exec -it <container-id> bash
